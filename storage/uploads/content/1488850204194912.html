SQL SELECT INTO

    
    Энэ нь хүснэгтийн бичлэгүүдийг өөр нэг шинэ хүснэгтэд хуулах зориулалттай SQL хэлний команд юм.

Хэрэглээ: Хүснэгтийг хуулж авах, backup үүсгэх. Мөн шинэчлэн оруулах, хүснэгт шинээр үүсгэх зэрэгт ашиглаж болно. Шинэчлэн оруулахдаа баганы нэрийг Alians ашиглан өөрчилж оруулж болно.

Бичигдэх хэлбэр /syntax/:
Хүснэгтийн тодорхой баганы бичлэгийг хуулах бол дараах хэлбэрээр бичнэ.
SELECT column_name1, column_name2, column_name2... INTO newtable  [IN externaldb]  FROM oldtable WHERE condition;;

Хүснэгтийн бүх баганы бичлэгүүдийг  хуулах бол дараах хэлбэрээр бичнэ.
SELECT * INTO newtable [IN externaldb] FROM oldtable WHERE condition;

Жишээ-1: Дараах query нь Customers хүснэгтийн бүх бичлэгүүдийг CustomersBackup2017 хүснэгтэд хуулж байна.

SELECT * INTO CustomersBackup2017 FROM Customers;

Жишээ-2: Дараах query нь Customers хүснэгтийн бүх бичлэгүүдийг IN оператор ашиглан Backup.mdb нэртэй өгөгдлийн сангийн CustomersBackup2017 хүснэгтэд хуулж байна.

SELECT * INTO CustomersBackup2017 IN 'Backup.mdb' FROM Customers;

Жишээ-3: Дараах query нь Customers хүснэгтийн CustomerName, ContactName баганы бичлэгүүдийг CustomersBackup2017 хүснэгтэд хуулж байна.

SELECT CustomerName, ContactName INTO CustomersBackup2017 FROM Customers;

Жишээ-4: Дараах query нь Customers хүснэгтийн Country баганы утга 'Germany' байх үеийн бүх бичлэгүүдийг CustomersGermany хүснэгтэд хуулж байна.

SELECT * INTO CustomersGermany FROM Customers WHERE Country = 'Germany';

Жишээ-5: Дараах query нь Customers болон Orders хүснэгтзүүдийн CustomerID талбараар нь LEFT JOIN хийж Хэрэглэгчийн нэр болон Захиалгын дугаар талбаруудын бичлэгүүдийг CustomersOrderBackup2017 нэртэй хүснэгтэд хуулж байна.

SELECT Customers.CustomerName, Orders.OrderID INTO CustomersOrderBackup2017 FROM Customers
LEFT JOIN Orders ON Customers.CustomerID = Orders.CustomerID;

Жишээ-6: Дараах query нь хуучин хүснэгтийн бүтцээр шинэ хүснэгт үүсгэж байна. 1 = 0 нь ямар ч өгөгдөлгүй үүсгэнэ гэдгийг зааж өгч байгаа юм.

SELECT * INTO newtable FROM oldtable WHERE 1 = 0;
