SQL CREATE TABLE

    
    Энэ нь өгөгдлийн санд шинээр хүснэгт үүсгэх SQL хэлний statement буюу команд юм.

Хэрэглээ: Тухайн нэг зүйлийг төлөөлөх бодит ертөнцийн юмс өгөгдлийн санд дүрслэгдэхдээ хүснэгт хэлбэрээр ордог. Өөрөөр хэлбэл өөрийн гэсэн шинж чанар бүхий зүйлс хүснэгтээр илэрхийлэгддэг. 

Бичигдэх хэлбэр /syntax/:
Шинээр үүсгэх
CREATE TABLE table_name (
    column1 datatype,
    column2 datatype,
    column3 datatype,
   ....
);

Өөр хүснэгт ашиглан үүсгэх
CREATE TABLE new_table_name AS
SELECT column1, column2,...
FROM existing_table_name
WHERE ....;

Жишээ-1: Өгөгдлийн сангийн хүснэгтийг дараах байдлаар бичиж үүсгэдэг.  

CREATE TABLE Persons (
    PersonID int,
    LastName varchar(255),
    FirstName varchar(255),
    Address varchar(255),
    City varchar(255)
);

Дээрх query бичиглэлд PersonID талбар нь INT төрлийн утга, бусад талбарууд нь VARCHAR буюу тэмдэгт мөр төрлийн утга авна гэдгийг тодорхойлж байна. Өгөгдлийн төрлийг тодорхойлохдоо мөн хэдэн тэмдэгтийн урттай байх вэ гэдгийг давхар зааж өгдөг.

Зураг 01 -с дээрх хүснэгт хэрхэн үүссэн байгааг харна уу.
