SQL PRIMARY KEY Constraint

    
    Энэ нь өгөгдлийн сангийн хүснэгтийн бичлэг бүрийг бүрэн төлөөлөх давтагдашгүй утгатайгаар тодорхойлогдсон түлхүүр юм. Өөрөөр хэлбэл тухайн мөр/бичлэгийг бусад бичлэгүүдээс ялгаж өгдөг анхдагч түлхүүр.

Хэрэглээ: Primary Key буюу анхдагч түлхүүр нь null утгатай байж болохгүй бөгөөд тухайн хүснэгт цор ганц анхдагч түлхүүртэй байдаг. Анхдагч түлхүүр талбар нь нэг болон тодорхой тооны талбаруудын нийлэмж байх боломжтой. Өөрөөр хэлбэл нэгээс дээш тооны баганаар анхдагч түлхүүр үүсгэж болно. 

Жишээ-1: Persons хүснэгтийг үүсгэхдээ ID баганыг Primary Key болгон тодорхойлох бол дараах байдлаар бичнэ. 

MySQL:
CREATE TABLE Persons (
    ID int NOT NULL,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255),
    Age int,
    PRIMARY KEY (ID)
);

SQL Server / Oracle / MS Access:
CREATE TABLE Persons (
    ID int NOT NULL PRIMARY KEY,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255),
    Age int    
);

Жишээ-2: Persons хүснэгтийг үүсгэхдээ Constraint командыг ашиглан ID болон LastName багануудыг Primary Key болгон тодорхойлох бол дараах байдлаар бичнэ.

MySQL / SQL Server / Oracle / MS Access:
CREATE TABLE Persons (
    ID int NOT NULL,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255),
    Age int,
    CONSTRAINT PK_Person PRIMARY KEY (ID, LastName)
);

Дээрх query -д CONSTRAINT командаар PK_Person нэртэй Primary key хязгаарлалтыг ID болон LastName баганууд дээр үүсгэхийг тодорхойлж өгсөн байна.

Жишээ-3: Тухайн хүснэгтийг аль хэдийн үүсгэсэн бол ALTER TABLE командаар засахдаа Constraint буюу хязгаарлалт тодорхойлж болно. Persons хүснэгтийг үүсгэсэн гэж үзээд ID талбарыг анхдагч түлхүүр болгож засахдаа дараах байдлаар тодорхойлж бичнэ. 

MySQL / SQL Server / Oracle / MS Access:
ALTER TABLE Persons ADD PRIMARY KEY (ID);

Constraint команд ашиглан 1 -с олон баганаар анхдагч түлхүүр үүсгэх бол дараах байдлаар бичиж болно.
ALTER TABLE Persons ADD CONSTRAINT PK_Person PRIMARY KEY (ID, LastName);

Жишээ-4: Тухайн үүсгэсэн Primary Key шаардлагагүй гэж үзээд ALTER TABLE DROP командын тусламжтай устгах бол дараах байдлаар бичнэ.

MySQL:
ALTER TABLE Persons DROP PRIMARY KEY;

SQL Server / Oracle / MS Access:
ALTER TABLE Persons DROP CONSTRAINT PK_Person;
