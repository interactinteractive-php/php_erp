SQL UPDATE

    
    Энэ нь хүснэгтэнд байгаа бичлэгийг өөрчлөхөд ашигладаг statement юм.

Бичигдэх хэлбэр /syntax/:

UPDATE table_name SET column_name1=value1, column_name2=value2, ... WHERE condition;

Хэрэглээ:Хүснэгтийн бичлэгийг өөрчилж байхдаа анхааралтай бай!. UPDATE statement -ийн WHERE clause -ийг сайн харж шалгаж бай. WHERE clause -р ямар бичлэг, хэр их бичлэг өөрчлөхийг тодорхойлж байдаг. Хэрэв WHERE clause орхигдох юмуу буруу бичигдвэл хүснэгтийн бүх бичлэг өөрчлөгдөх аюултай!

Зураг 01 Бид доорх хүснэгтийг жишээндээ ашиглана..
    
     
  
                                                                                                                                                                                                                                   Зураг 01  Products хүснэгт

    
    Жишээ-1: Products хүснэгтийн эхний бичлэгийг буруу оруулсан байна. Бид бичлэгийг засах хэрэгтэй боллоо үзье.

UPDATE Products SET ProductName = 'Blossom Tea', Price = 20 WHERE ProductID = 1;

Зураг 02 -с дээрх UPDATE query хэрхэн ажилласан байгааг харна уу.
    
   
                                                                                                                                                                                             Зураг 02 ProductName, Price өөрчлөгдсөн байна.		    
    Жишээ-2: Product хүснэгтээс нэг дор олон бичлэг солиж үзье. 

UPDATE  Products SET CategoryID=3 WHERE CategoryID=2;

Зураг 03 -с дээрх query хэрхэн ажилласан байгааг харна уу.
    
   
                                                                                                                                                                                                     Зураг 03 CategoryID -ийг өөрчилсөн байна.		    
    Зурагт CategoryID баганаас дөрвөлжилж тодоруулсан 3 бичлэг өөрчлөгдсөн байгааг харж байна. WHERE clause -ийн тусламжтай CategoryID нь 2 той тэнцүү бичлэгийг 3 болгон солисон байна. 

Анхааруулга!
Дахин сануулахад WHERE clause -г зааж өгөөгүй бол хүснэгтийн бичлэг тэр чигээрээ өөрчлгдөх аюултай.
