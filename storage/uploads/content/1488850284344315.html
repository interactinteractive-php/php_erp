SQL CREATE DATABASE

    
    Энэ нь ӨСУС -д өгөгдлийн сан шинээр үүсгэх SQL хэлний statement буюу команд юм.

Хэрэглээ: Хамгийн энгийн дээ бол өгөгдлийн сан нь зүгээр л файл юм. Энэ нь тодорхой логик бүтэцтэй байдаг бөгөөд гол агуулах мэдээллүүд нь бидний мэдэхээр хүснэгт юм. Өгөгдлийн сан удирдах системд өгөгдлийн сан нь тодорхой үүрэг зориулалтын дагуу өгөгдөл оруулж хадгалах ерөнхий хэрэглээтэй байдаг тул програм хангамж, систем хийхдээ шинээр үүсгэж ашиглах зайлшгүй шаардлага гардаг. 

Бичигдэх хэлбэр /syntax/:
Богино хэлбэр
CREATE DATABASE databasename;

Дэлгэрэнгүй хэлбэр

CREATE DATABASE databasename CHARACTER SET charset_name COLLATE collation_name;

Жишээ-1: Өгөгдлийн сан үүсгэхдээ DEFAULT байдлаар дараах байдлаар үүсгэдэг.  

CREATE DATABASE testDB;

Дээрх query бичиглэлээр өгөгдлийн сан үүсгэвэл бүх тохиргоо Default буюу анх тодорхойлогдсон байдлаар үүсэх бөгөөд өгөгдлийн сан удирдах системүүдэд асуудал үүсэхгүй юм.

Жишээ-2: Өгөгдлийн санг илүү дэлгэрэнгүй тодорхой байдлаар үүсгэх жишээг харцгаая.
CREATE DATABASE testDB DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci ;

Дээрх query бичиглэлээр үүсгэвэл өгөгдлийн сан нь UTF-8 тэмдэгтийн формат буюу бүлийг дэмжиж ажиллах юм.
