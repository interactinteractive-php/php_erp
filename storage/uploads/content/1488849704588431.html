SQL INSERT INTO

    
    Энэ нь өгөгдлийн сангийн хүснэгтэд шинээр өгөгдөл буюу бичлэг нэмэх зорилгоор ашигладаг SQL хэлний statement буюу команд юм.

Хэрэглээ: Өгөгдлийн сангийн хүснэгтүүдэд бичлэг нэмэх үйлдлийг INSERT query -р гүйцэтгэдэг бөгөөд үндсэн хоёр хэлбэрээр бичдэг.

Бичигдэх хэлбэр /syntax/:
1. Өгөгдөл оруулах багануудын нэрийг тодорхойлж бичих
    INSERT INTO table_name (column1, column2, column3, ...) VALUES (value1, value2, value3, ...);

2. Өгөгдөл оруулах хүснэгтийн нэрийг тодорхойлж бичих
    INSERT INTO table_name VALUES (value1, value2, value3, ...);

Жишээ-1: Хэрэглэгчдийн мэдээллийн жагсаалтанд бичлэг нэмэх шаардлага гарсан гэж үзье.
SELECT * FROM Customers;

Зураг 01 -с бичлэг нэмээгүй байгааг Customers хүснэгтийг харна уу.		    
                                                                                                                                                                                                                                 Зураг 01 хэрэглэгчдийн жагсаалт    
   
Зурагт INSERT query ажиллуулахаас өмнөх Customers хүснэгтийн мэдээллийг SELECT query ашиглан харуулсан байна. Энд * тэмдэгт нь тухайн хүснэгтийн бүх талбарын select хийж байгааг илэрхийлнэ.

INSERT INTO Customers (CustomerName, ContactName, Address, City, PostalCode, Country)
VALUES ('Bólido Comidas preparadas','Martín Sommer','C/ Araquil, 67','Madrid','28023','Spain');

SELECT * FROM Customers;

Зураг 02 -с дээрх INSERT query ажиллуулсаны дараах SELECT query хэрхэн ажилласан байгааг харна уу.

    
   
                                                                                                                                                                    Зураг 02 Германы Берлин хотод амьдардаг хэрэглэгчийн мэдээлэл		    
   
Зурагт INSERT командын хэрэглээг харуулсан бөгөөд Customers хүснэгтэд 8 дугаартай бичлэг нэмэгдсэн байгааг харуулсан байна. Энд тэмдэгт мөр /varchar/ төрлийн өгөгдлийг дан хашилтан дунд буюу 'data' байдлаар бичнэ.

Жишээ-2: Хэрэв хүснэгтийн багана бүрт харгалзах утга оруулах шаардлагагүй бол тодорхой багануудын нэрийг тодорхойлж бичлэг нэмэх үйлдлийг гүйцэтгэдэг.

INSERT INTO Customers (CustomerName, City, Country) VALUES ('Cardinal','Stavanger','Norway');

SELECT * FROM Customers;

Зураг 03 -с дээрх INSERT query ажиллуулсаны дараах SELECT query хэрхэн ажилласан байгааг харна уу.
    
    Зураг 02 Германы Берлин хотод амьдардаг хэрэглэгчийн мэдээлэл    
   
Зурагт INSERT командын хэрэглээг харуулсан бөгөөд Customers хүснэгтийн зөвхөн CustomerName, City, Country зэрэг багануудад өгөгдөл нэмж харуулсан байна. Энд INSERT query -д тодорхойлоогүй багануудад харгалзах утга NULL утгатай орно. NULL хоосон буюу ямар ч утга ороогүйг илтгэх юм.
