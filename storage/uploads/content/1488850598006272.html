SQL AUTO INCREMENT

    
    Энэ нь тухайн тоон төрлийн утгатай баганы өгөгдөл хүснэгтэд бичлэг нэмэгдэх бүрт өгөгдсөн зүй тогтолын дагуу автоматаар нэмэгдэж орох арга хэрэгсэл юм. 

Хэрэглээ: Ихэвчлэн Primary Key буюу анхдагч түлхүүр талбарт Auto Increment -г ашигладаг. Энэ нь бичлэг нэмэгдэх бүрт анхдагч түлхүүр нь автоматаар үүсэх боломж олгоно. 

Жишээ-1: Auto Increment -г MySQL -д хэрхэн  хэрэгжүүлэхийг харцгаая.

CREATE TABLE Persons (
    ID int NOT NULL AUTO_INCREMENT,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255),
    Age int,
    PRIMARY KEY (ID)
);

MySQL нь AUTO_INCREMENT түлхүүр үгээр тухайн баганыг тохируулдаг бөгөөд дээрх query -д Persons хүснэгтийн анхдагч түлхүүр болох ID талбарын утгыг автоматаар нэмэгдэнэ гэдгийг тодорхойлсон байна. Энд ID талбарын утга 1-с эхлэн нэг нэгээр нэмэгдэж орох юм. Харин эхлэлийн утгыг тодорхойлохдоо дараах байдлаар бичнэ.

ALTER TABLE Persons AUTO_INCREMENT=100;

Дээрх query -д ALTER TABLE командын тусламжтай эхлэлийн утгыг 100 гэж тодорхойлсон байна. Нэгэнт автоматаар нэмэгдэж явах учираас бичлэг нэмэхдээ auto increment -р үүсгэсэн анхдагч түлхүүр талбарын утгыг оруулах шаардлагагүйг дараах Insert query -с харна уу. 
 
INSERT INTO Persons (FirstName, LastName) VALUES ('Lars','Monsen');

Жишээ-2: Auto Increment -г SQL SERVER -д хэрхэн  хэрэгжүүлэхийг харцгаая.

CREATE TABLE Persons (
    ID int IDENTITY(1, 1) PRIMARY KEY,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255),
    Age int
);

SQL Server нь IDENTITY түлхүүр үгээр тухайн баганыг тохируулдаг бөгөөд дээрх query -д Persons хүснэгтийн анхдагч түлхүүр болох ID талбарын утгыг автоматаар нэмэгдэнэ гэдгийг тодорхойлсон байна. Энд ID талбарын утга 1-с эхлэн нэг нэгээр нэмэгдэж орох юм. Хэрэв эхлэлийн утга болон нэмэгдэх утгыг өөрчлөх бол IDENTITY(10, 5) байдлаар бичиж болно. Энд 10 -с эхлэн 5 -р нэмэгдэж орно гэдгийг илэрхийлж байна.

Жишээ-3: Auto Increment -г MS ACCESS -д хэрхэн  хэрэгжүүлэхийг харцгаая.

CREATE TABLE Persons (
    ID int PRIMARY KEY AUTOINCREMENT,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255),
    Age int
);

MS ACCESS нь AUTOINCREMENT түлхүүр үгээр тухайн баганыг тохируулдаг бөгөөд дээрх query -д Persons хүснэгтийн анхдагч түлхүүр болох ID талбарын утгыг автоматаар нэмэгдэнэ гэдгийг тодорхойлсон байна. Энд ID талбарын утга 1-с эхлэн нэг нэгээр нэмэгдэж орох юм. Хэрэв эхлэлийн утга болон нэмэгдэх утгыг өөрчлөх бол AUTOINCREMENT(10, 5) байдлаар бичиж болно. Энд 10 -с эхлэн 5 -р нэмэгдэж орно гэдгийг илэрхийлж байна.

Жишээ-2: Auto Increment -г ORACLE -д хэрхэн  хэрэгжүүлэхийг харцгаая. 

CREATE SEQUENCE seq_person
MINVALUE 1
START WITH 1
INCREMENT BY 1
CACHE 10;

ORACLE нь бусад ӨСУС -с ялгаатай байдлаар хэрэгжүүлдэг бөгөөд дараалал бүтэцтэй обьект үүсгэж хадгалдаг. Sequence Object нь үүсгэсэн бүтцийнхээ дагуу тоон утгыг автоматаар гаргадаг.
Дээрх кодын хэсэгт 1-с эхлэн нэгээр нэмэгдэж үүсэхийг тодорхойлж өгсөн бөгөөд CACHE нь эхний 10 -н утгыг кээш санах ой руу хийж тухайн нэмэгдсэн утгыг дуудах үед хурдан буцаах үүрэгтэй юм. 
 
Харин бичлэг нэмэхдээ дээрх seq_person нэртэй дараалал бүтэцтэй обьектоос утгаа nextVal() функцийн тусламжтай авч оруулах юм. 

INSERT INTO Persons (ID,FirstName,LastName) VALUES (seq_person.nextval,'Lars','Monsen');
