SQL ALTER TABLE

    
    Энэ нь өгөгдлийн сангийн хүснэгт засахад ашиглагддаг SQL хэлний statement буюу команд юм. 

Хэрэглээ: Өгөгдлийн сангийн хүснэгтэд багана нэмэх, устгах, засварлах үед ашиглана. Дараах жагсаалтаас Alter Table командын хэрэглээг харна уу. 
  • ALTER TABLE -> ADD COLUMN - хүснэгтэд багана нэмэх 
  • ALTER TABLE -> DROP COLUMN - хүснэгтээс багана устгах
  • ALTER TABLE -> ALTER/MODIFY COLUMN - хүснэгтийн баганы өгөгдлийн төрлийг өөрчилөх
Бичигдэх хэлбэр /syntax/:
ADD COLUMN
ALTER TABLE table_name ADD column_name datatype;

DROP COLUMN
ALTER TABLE table_name DROP COLUMN column_name;

ALTER/MODIFY COLUMN
  • SQL SERVER / MS ACCESS - ALTER TABLE table_name ALTER COLUMN column_name datatype;
  • MYSQL / ORACLE 10G өмнөх - ALTER TABLE table_name MODIFY COLUMN column_name datatype;
  • ORACLE 10G дараах - ALTER TABLE table_name MODIFY column_name datatype;
Зураг 01 -с жишээнд ашиглагдах Person хүснэгтийг харна уу.		    
     
                                                                                                                                                                                                        Зураг 01 Person хүснэгт		    
    Жишээ-1: Дээрх Person хүснэгтэд төрсөн огноог авах шинэ багана нэмэх шаардлагатай болсон гэж үзье.

ALTER TABLE Persons ADD DateOfBirth date;

Зураг 02 -с дээрх ALTER query хэрхэн ажилласан байгааг харна уу.
    
   
                                                                                                                       Зураг 02 Person хүснэгтэд DateOfBirth нэртэй багана нэмэгдсэн байдал		    
   
Зурагт ALTER TABLE ADD Column командын үр дүнг харуулсан бөгөөд Person хүснэгтэд DATE төрөлтэй DateOfBirth нэртэй шинэ багана нэмэгдсэн байна. 

Жишээ-2: Person хүснэгтийн DateOfBirth баганы өгөгдлийн төрлийг өөрчлөх шаардлага гарсан гэж үзье.

ALTER TABLE Persons ALTER COLUMN DateOfBirth year;

    
    Дээрх query -д DateOfBirth баганы өгөгдлийн төрлийн YEAR болгон өөрчилж байна.

Жишээ-3: Person хүснэгтээс DateOfBirth баганыг устгах шаардлага гарсан гэж үзье.

ALTER TABLE Persons DROP COLUMN DateOfBirth;

Зурагт ALTER TABLE DROP COLUMN  командын ажилласан үр дүнг харуулсан бөгөөд Person хүснэгтээс DateOfBirth багана устсан байна.
