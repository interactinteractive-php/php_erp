SQL INSERT INTO SELECT

    
    Энэ нь нэг хүснэгтээс өгөгдлүүдийг хуулж аваад өөр нэг хүснэгтэд нэмж оруулдаг SQL хэлний statement буюу команд юм.

Хэрэглээ: Хүснэгт хооронд бичлэг эсвэл тодорхой өгөгдлүүдийг хуулж хэрэгцээтэй байгаа хүснэгтэд нэмэх шаардлагатай үед ашиглана. Гол шаардлага нь нэмж хуулах хүснэгтийн өгөгдлийн төрөл таарч байх мөн өмнө нь нэмэгдэж ороогүй бичлэг байх гэх зэрэг зарчимтай.

Бичигдэх хэлбэр /syntax/:
Хүснэгтийн бүх баганы бичлэгийг нэмж хуулах бол
 INSERT INTO table2 SELECT * FROM table1 WHERE condition;

Хүснэгтийн тодорхой багануудын нэрийг тодорхойлж нэмж хуулах
INSERT INTO table2 (column1, column2, column3, ...) SELECT column1, column2, column3, ... FROM table1 WHERE condition;

Жишээ-1: Хэрэглэгчдийн мэдээллийн жагсаалтанд бичлэг нэмэхдээ Suppliers хүснэгтээс харгалзах баганыг тодорхойлж оруулах жишээ авч үзье.

INSERT INTO Customers (CustomerName, City, Country) SELECT SupplierName, City, Country FROM Suppliers;

    
    Дээрх query -д Suppliers хүснэгтийн SupplierName, City, Country талбаруудын утгыг Customers хүснэгтийн CustomerName, City, Country талбаруудад нэмж оруулж байна. Утга оруулахаар тодорхойлж өгөөгүй баганын утга NULL -аар орно.

    
    Жишээ-2: Customers хүснэгтэд Suppliers хүснэгтээс бичлэг нэмж оруулахдаа Country талбарын утга нь 'Germany' байх бичлэгүүдийг оруулах жишээ авч үзье.

INSERT INTO Customers (CustomerName, City, Country) SELECT SupplierName, City, Country FROM Suppliers WHERE Country='Germany';

Зураг 01 -с дээрх INSERT query хэрхэн ажилласан байгааг харна уу.

    
   
Зураг 01 Country баганы утга Germany байх үеийн бүх бичлэг Customers хүснэгтэд нэмэгдсэн харагдац		    
   
Зурагт өгөгдөл оруулж өгөхөөр торхойлсон багануудаас бусад баганы утга NULL байгааг харуулсан байна. Мөн сүүлд нэмэгдсэн бичлэгүүдийн Country баганы утга Germany байна.
